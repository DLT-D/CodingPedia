Article 4

-- Create a table
CREATE TABLE students (
    id INT PRIMARY KEY,
    name VARCHAR(50),
    age INT
);

-- Insert data into the table
INSERT INTO students (id, name, age)
VALUES (1, 'Alice', 21),
       (2, 'Bob', 22);

-- Select all data
SELECT * FROM students;

-- Select only names of students older than 21
SELECT name FROM students WHERE age > 21;

-- Update a record
UPDATE students
SET age = 23
WHERE name = 'Alice';

-- Delete a record
DELETE FROM students
WHERE id = 2;
